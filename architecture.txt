🛡️ Causal-Guard: Complete System Architecture
Executive Summary
Causal-Guard is a production-grade neuro-symbolic validation layer that intercepts LLM-generated explanations and verifies them against five causal constraints (C₁–C₅) before they reach decision-makers. The system is designed for high throughput, low latency, and explainable failures, making it suitable for integration into urban logistics platforms, traffic management systems, and other safety-critical applications.

This document provides a complete architectural specification from data ingestion to deployment, including every component, interface, data structure, and operational consideration.

1. System Overview
1.1 Purpose
Causal-Guard validates LLM-generated explanations for urban transportation incidents against five causal admissibility constraints:

Constraint	Description	Validation Method
C₁: Temporal Precedence	Cause must precede effect	Rule-based temporal reasoning
C₂: Spatial Relevance	Cause must be geographically plausible	Graph-based spatial reasoning
C₃: Mechanistic Plausibility	Causal mechanism must be physically possible	Hybrid (embedding + rules)
C₄: Non-Spuriousness	Must avoid correlational fallacies	Statistical + rule-based
C₅: Completeness	Must include all necessary factors	Template matching + ML
1.2 Design Goals
Goal	Target	Rationale
Latency	<500ms per explanation	Real-time decision support
Throughput	100+ requests/second	Production scale
Precision	>90%	Avoid false alarms
Recall	>80%	Catch most violations
Explainability	Human-readable violation reasons	Trust and debugging
Extensibility	Plugin architecture	Add new checkers easily
Deployability	Docker + Kubernetes	Cloud/on-prem flexibility

request_id must be UUID v4

scenario.id must exist in scenario registry

config.checkers must be subset of ["C1","C2","C3","C4","C5"]

class GroqProvider(LLMProvider):

        self.model = model

            max_tokens=max_tokens

            "explanation": response.choices[0].message.content,
            "model": self.model,
            "latency_ms": response.usage.total_time,
            "tokens": {
                "prompt": response.usage.prompt_tokens,
                "completion": response.usage.completion_tokens
            }
        }

# orchestrator.py
class CausalGuardOrchestrator:
    def __init__(self, config: dict):
        self.checkers = {
            'C1': TemporalChecker(config.get('c1_config', {})),
            'C2': SpatialChecker(config.get('c2_config', {})),
            'C3': MechanismChecker(config.get('c3_config', {})),
            'C4': SpuriousChecker(config.get('c4_config', {})),
            'C5': CompletenessChecker(config.get('c5_config', {}))
        }
        self.kb = KnowledgeBase(config['kb_path'])
        self.road_graph = RoadGraphService(config['graph_db_url'])
        self.cache = CacheService(config['redis_url'])
        self.metrics = MetricsCollector()
    
    async def validate(
        self, 
        request_id: str,
        scenario: dict,
        explanation: str,
        enabled_checkers: list
    ) -> dict:
        """Run enabled checkers and aggregate results"""
        
        start_time = time.time()
        results = {}
        violations = []
        
        # Check cache first
        cached = await self.cache.get(f"validation:{request_id}")
        if cached:
            return cached
        
        # Run checkers in parallel
        tasks = []
        for checker_id in enabled_checkers:
            if checker_id in self.checkers:
                tasks.append(
                    self._run_checker(
                        checker_id, 
                        self.checkers[checker_id],
                        scenario, 
                        explanation
                    )
                )
        
        checker_results = await asyncio.gather(*tasks)
        
        # Aggregate results
        for result in checker_results:
            results[result['checker_id']] = result
            if not result['passed']:
                violations.append({
                    'constraint': result['checker_id'],
                    'reason': result['reason'],
                    'details': result.get('details', {})
                })
        
        # Determine overall status
        status = 'PASS' if len(violations) == 0 else 'FAIL'
        
        output = {
            'request_id': request_id,
            'timestamp': datetime.utcnow().isoformat(),
            'status': status,
            'violations': violations,
            'results': results,
            'latency_ms': int((time.time() - start_time) * 1000),
            'checkers_run': enabled_checkers
        }
        
        # Cache result
        await self.cache.set(
            f"validation:{request_id}", 
            output, 
            ttl=3600
        )
        
        # Collect metrics
        self.metrics.record(request_id, output)
        
        return output
    
    async def _run_checker(self, checker_id, checker, scenario, explanation):
        try:
            result = await checker.check(scenario, explanation)
            return {
                'checker_id': checker_id,
                'passed': result['passed'],
                'reason': result.get('reason', ''),
                'details': result.get('details', {}),
                'latency_ms': result.get('latency_ms', 0)
            }
        except Exception as e:
            # Log error, return as violation
            return {
                'checker_id': checker_id,
                'passed': False,
                'reason': f'Checker error: {str(e)}',
                'error': True
            }

# checkers/c1_temporal.py
import re
from dateutil import parser
from datetime import datetime

class TemporalChecker:
    def __init__(self, config):
        self.time_extractor = TimeExtractor()
        self.tolerance_seconds = config.get('tolerance_seconds', 60)
    
    async def check(self, scenario: dict, explanation: str) -> dict:
        start = time.time()
        
        # Step 1: Extract all causal claims from explanation
        claims = self._extract_causal_claims(explanation)
        
        # Step 2: Extract timeline from scenario
        timeline = self._build_timeline(scenario)
        
        violations = []
        
        for claim in claims:
            cause_time = self._find_event_time(claim['cause'], timeline)
            effect_time = self._find_event_time(claim['effect'], timeline)
            
            if cause_time and effect_time:
                # Check temporal precedence
                if cause_time >= effect_time:
                    violations.append({
                        'claim': f"{claim['cause']} caused {claim['effect']}",
                        'cause_time': cause_time.isoformat(),
                        'effect_time': effect_time.isoformat(),
                        'reason': f"Cause occurs after effect"
                    })
            elif cause_time and not effect_time:
                # Effect not in timeline - could be derived
                pass
            elif not cause_time and effect_time:
                # Cause not in timeline - could be inferred
                pass
        
        passed = len(violations) == 0
        
        return {
            'passed': passed,
            'reason': 'All temporal relationships valid' if passed else f'{len(violations)} temporal violation(s)',
            'details': {
                'claims_extracted': len(claims),
                'violations': violations
            },
            'latency_ms': int((time.time() - start) * 1000)
        }
    
    def _extract_causal_claims(self, text: str) -> list:
        """Extract (cause, effect) pairs using dependency parsing"""
        # Pattern-based extraction for MVP
        patterns = [
            (r'(.+?)\s+caused\s+(.+)', 'caused'),
            (r'(.+?)\s+led to\s+(.+)', 'led to'),
            (r'(.+?)\s+resulted in\s+(.+)', 'resulted in'),
            (r'due to\s+(.+?),\s+(.+)', 'due to'),
        ]
        
        claims = []
        for pattern, rel in patterns:
            matches = re.findall(pattern, text)
            for match in matches:
                claims.append({
                    'cause': match[0].strip(),
                    'effect': match[1].strip(),
                    'relation': rel
                })
        
        return claims
    
    def _build_timeline(self, scenario: dict) -> dict:
        """Build timeline of events from scenario context"""
        timeline = {}
        for event in scenario.get('context', {}).get('timeline', []):
            time = parser.parse(event['time'])
            timeline[event['event']] = {
                'time': time,
                'description': event.get('description', ''),
                'value': event.get('value')
            }
        return timeline
    
    def _find_event_time(self, text: str, timeline: dict) -> datetime:
        """Find the time of an event mentioned in text"""
        # Simple keyword matching for MVP
        for event_name, event_data in timeline.items():
            if event_name.lower() in text.lower():
                return event_data['time']
        
        # Try to extract absolute time from text
        time_match = re.search(r'(\d{1,2}):(\d{2})\s*(AM|PM)?', text)
        if time_match:
            # Parse time (simplified)
            hour = int(time_match.group(1))
            minute = int(time_match.group(2))
            meridian = time_match.group(3)
            if meridian and meridian.upper() == 'PM' and hour < 12:
                hour += 12
            # Assume same date as scenario
            scenario_date = datetime.now().date()  # Should come from scenario
            return datetime.combine(scenario_date, datetime.min.time().replace(hour=hour, minute=minute))
        
        return None

# checkers/c2_spatial.py
import re
import aiohttp
from math import radians, sin, cos, sqrt, atan2

class SpatialChecker:
    def __init__(self, config):
        self.graph_service_url = config['graph_service_url']
        self.max_distance_km = config.get('max_distance_km', 2.0)
        self.require_network_connectivity = config.get('require_network_connectivity', True)
    
    async def check(self, scenario: dict, explanation: str) -> dict:
        start = time.time()
        
        # Step 1: Extract location entities from explanation
        mentioned_locations = self._extract_locations(explanation)
        
        # Step 2: Get scenario locations
        scenario_locations = scenario.get('context', {}).get('locations', [])
        
        # Step 3: Check each mentioned location
        violations = []
        
        for loc in mentioned_locations:
            # Find matching scenario location
            matched = False
            for scenario_loc in scenario_locations:
                if self._locations_match(loc, scenario_loc):
                    matched = True
                    break
            
            if not matched:
                # Check if this location is plausible
                plausible = await self._check_plausibility(
                    loc, 
                    scenario_locations,
                    scenario.get('context', {}).get('road_network', {})
                )
                
                if not plausible['is_plausible']:
                    violations.append({
                        'location': loc['text'],
                        'reason': plausible['reason'],
                        'distance_km': plausible.get('distance_km')
                    })
        
        passed = len(violations) == 0
        
        return {
            'passed': passed,
            'reason': 'All locations spatially relevant' if passed else f'{len(violations)} spatial violation(s)',
            'details': {
                'mentioned_locations': mentioned_locations,
                'violations': violations
            },
            'latency_ms': int((time.time() - start) * 1000)
        }
    
    def _extract_locations(self, text: str) -> list:
        """Extract location mentions using NER and patterns"""
        # Simple pattern-based extraction for MVP
        patterns = [
            r'on\s+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)',
            r'at\s+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)',
            r'near\s+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)',
        ]
        
        locations = []
        for pattern in patterns:
            matches = re.findall(pattern, text)
            for match in matches:
                locations.append({
                    'text': match,
                    'pattern': pattern
                })
        
        return locations
    
    def _locations_match(self, mentioned: dict, scenario_loc: dict) -> bool:
        """Check if mentioned location matches scenario location"""
        mentioned_text = mentioned['text'].lower()
        scenario_name = scenario_loc.get('name', '').lower()
        
        # Exact match
        if mentioned_text == scenario_name:
            return True
        
        # Partial match (e.g., "SZR" matches "Sheikh Zayed Road")
        if mentioned_text in scenario_name or scenario_name in mentioned_text:
            return True
        
        return False
    
    async def _check_plausibility(self, loc: dict, scenario_locs: list, road_network: dict) -> dict:
        """Check if a mentioned location is plausible given scenario context"""
        
        # If we have coordinates, calculate distance to nearest scenario location
        if 'coordinates' in loc:
            min_distance = float('inf')
            nearest = None
            
            for s_loc in scenario_locs:
                if 'coordinates' in s_loc:
                    dist = self._haversine_distance(
                        loc['coordinates'],
                        s_loc['coordinates']
                    )
                    if dist < min_distance:
                        min_distance = dist
                        nearest = s_loc
            
            if min_distance <= self.max_distance_km:
                return {
                    'is_plausible': True,
                    'distance_km': min_distance,
                    'reason': f'Within {min_distance:.2f}km of {nearest["name"]}'
                }
            else:
                return {
                    'is_plausible': False,
                    'distance_km': min_distance,
                    'reason': f'Location is {min_distance:.2f}km from nearest scenario location'
                }
        
        # Without coordinates, assume plausible but flag for review
        return {
            'is_plausible': True,
            'reason': 'No coordinates available for verification'
        }
    
    def _haversine_distance(self, coord1, coord2):
        """Calculate great-circle distance between two coordinates"""
        lat1, lon1 = radians(coord1[0]), radians(coord1[1])
        lat2, lon2 = radians(coord2[0]), radians(coord2[1])
        
        dlon = lon2 - lon1
        dlat = lat2 - lat1
        
        a = sin(dlat/2)**2 + cos(lat1) * cos(lat2) * sin(dlon/2)**2
        c = 2 * atan2(sqrt(a), sqrt(1-a))
        
        # Earth radius in km
        R = 6371
        return R * c

# checkers/c3_mechanism.py
import numpy as np
from sentence_transformers import SentenceTransformer
import faiss
import json
import re

class MechanismChecker:
    def __init__(self, config):
        self.model = SentenceTransformer(config.get('embedding_model', 'all-MiniLM-L6-v2'))
        self.kb_path = config['kb_path']
        self.similarity_threshold = config.get('similarity_threshold', 0.7)
        self.top_k = config.get('top_k', 5)
        
        # Load knowledge base
        self.kb = self._load_knowledge_base()
        
        # Build FAISS index
        self.index = self._build_index()
        
        # Rule-based overrides for threshold cases
        self.rules = self._load_rules()
    
    def _load_knowledge_base(self):
        with open(self.kb_path, 'r') as f:
            data = json.load(f)
        return data['mechanisms']
    
    def _build_index(self):
        # Encode all mechanism descriptions
        descriptions = [m['description'] for m in self.kb]
        embeddings = self.model.encode(descriptions)
        
        # Build FAISS index
        dimension = embeddings.shape[1]
        index = faiss.IndexFlatIP(dimension)  # Inner product for cosine similarity
        index.add(embeddings)
        
        return index
    
    def _load_rules(self):
        """Load rule-based overrides for specific mechanisms"""
        return [
            {
                'name': 'black_ice_impossible',
                'pattern': r'black ice.*?(?:formed|developed).*?(\d+)°C',
                'condition': lambda match: int(match.group(1)) > 0,
                'reason': 'Black ice cannot form above 0°C'
            },
            {
                'name': 'hydroplaning_requires_water',
                'pattern': r'hydroplan(?:ed|ing).*?(?:without|no) (?:water|rain|standing water)',
                'condition': lambda match: True,
                'reason': 'Hydroplaning requires standing water'
            }
        ]
    
    async def check(self, scenario: dict, explanation: str) -> dict:
        start = time.time()
        
        # Step 1: Extract mechanism description from explanation
        mechanism_text = self._extract_mechanism(explanation)
        
        if not mechanism_text:
            return {
                'passed': True,
                'reason': 'No mechanism described to verify',
                'latency_ms': int((time.time() - start) * 1000)
            }
        
        # Step 2: Check rule-based overrides first
        for rule in self.rules:
            match = re.search(rule['pattern'], explanation, re.IGNORECASE)
            if match and rule['condition'](match):
                return {
                    'passed': False,
                    'reason': rule['reason'],
                    'details': {
                        'rule': rule['name'],
                        'matched_text': match.group(0)
                    },
                    'latency_ms': int((time.time() - start) * 1000)
                }
        
        # Step 3: Embed and retrieve similar mechanisms
        embedding = self.model.encode([mechanism_text])
        distances, indices = self.index.search(embedding, self.top_k)
        
        # Step 4: Check similarity threshold
